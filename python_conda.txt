
# --- instalar o miniconda

# --- inclui e define conda-forge como canal prioritario
conda config --add channels conda-forge
conda config --set channel_priority strict

# -- atualiza conda
conda update conda

# --- comandos de consulta geral
# ver informacoes
conda info

# ver ambientes virtuais
conda info --envs

# listar pacotes/biblitoecas
conda list

# instalar pacotes
conda install <package-name>

# --- para criar um ambiente novo, chamado 'myenv'

# exemplo 1: com python 3.7 exatamente (==)

# exemplo 2: com python 3.7.x mais atual

# exemplo 3: com python 3 mais atual

# --- para ativar o ambiente
conda activate myenv

# --- para desativar o ambiente atual
conda deactivate
